<scope> ::= '{' <statement_list> '}'
<statement_list> ::= <statement> | <statement> <statement_list>
<statement> ::= <declaration> | <reassignment> | <print_statement> | <exit_statement> | <function_declaration>
<declaration> ::= <type> <identifier> = <expression>;
<reassignment> ::= <identifier> = <expression>;
<print_statement> ::= print [<str_expression> | <char_expression>];
<exit_statement> ::= exit <num_expression>;
<function_declaration> ::= <type> <identifier>'(' <parameter>, <parameter>, ... ')' [<scope> | ';']
<function_call> ::= <identifier>'(' <parameter>, <parameter>, ... ')'
<function_call_stmt> ::= <function_call>;
<return_statement> ::= return <expression>;

<expression> ::= <num_expression> | <char_expression> | <str_expression>

<num_expression> ::= <num_term> | <num_expression> ('+' | '-') <num_expression>
<num_term> ::= <num_factor> | <num_term> ('*' | '/' | '%') <num_term>
<num_factor> ::= <num_literal> | <char_literal> | <identifier> | | <function_call> | '(' <num_expression> ')'

<char_expression> ::= <num_expression>
<str_expression> ::= <str_literal> | <identifier> | <function_call>

<num_literal> ::= [0-9]+
<char_literal> ::= '*'
<str_literal> ::= "*"
<identifier> ::= [a-zA-Z][a-zA-Z0-9_]*
<type> ::= int | float | str | char
